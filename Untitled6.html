#define a function which resize our data to the size of the dem raster, it doesn't change resolution so we need to have
#the same resolution on each entry raster, but I will improve it. It doesn't "moove" our data more than one unit of resolution
def resize_raster (raster_dem,r_to_resize):
    I=raster_dem.ReadAsArray()
    shape=I.shape
    r_array=r_to_resize.ReadAsArray()
    
    I=I*0 #we will put our resized raster in this I raster of the right size
    gt_dem = raster_dem.GetGeoTransform()
    gt_r = r_to_resize.GetGeoTransform()
    

    new_raster_shape = (r_array.shape[0]/gt_dem[5], r_array.shape[1]/gt_dem[1])
    
    ncols_offset = round((gt_r[0] - gt_dem[0]) / gt_dem[1])
    nrows_offset = round((gt_r[3] - gt_dem[3]) / gt_dem[5])
    I[ncols_offset:(ncols_offset+r_array.shape[0]),nrows_offset:(nrows_offset+r_array.shape[1])] = r_array
    l,c=0,0
    for i in I:
        l=l+1
        c=0
        for k in i:
            c=c+1
            if abs(k)>10**6:
                I[l-1,c-1]=0
    return I

def write_geotiff(raster, gt, wkt, outputpath, dtype=gdal.GDT_UInt16, options=['COMPRESS=LZW'], color_table=0, nbands=1, nodata=False):
    width = raster.shape[1]
    height =raster.shape[0]

    # Prepare destination file
    driver = gdal.GetDriverByName("GTiff")
    if options != 0:
        dest = driver.Create(outputpath, width, height, nbands, dtype, options)
    else:
        dest = driver.Create(outputpath, width, height, nbands, dtype)

    # Write output raster
    if color_table != 0:
        dest.GetRasterBand(1).SetColorTable(color_table)

    dest.GetRasterBand(1).WriteArray(raster)

    if nodata is not False:
        dest.GetRasterBand(1).SetNoDataValue(nodata)

    # Set transform and projection
    dest.SetGeoTransform(gt)
    srs = osr.SpatialReference()
    srs.ImportFromWkt(wkt)
    dest.SetProjection(srs.ExportToWkt())

    # Close output raster dataset 
    dest = None
